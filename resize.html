let resizeTimeout; let lastWidth = window.innerWidth; let lastHeight = window.innerHeight; window.addEventListener("resize", function() { const currentWidth = window.innerWidth; const currentHeight = window.innerHeight; // KEYPAD DETECTION: If the height increases significantly while width stays the same, // it means the mobile keyboard was closed. if (currentWidth === lastWidth && currentHeight > lastHeight) { // Find our text/number input and strip focus to remove the blinking cursor const activeInput = document.activeElement; if (activeInput && (activeInput.type === 'number' || activeInput.type === 'text')) { activeInput.blur(); } } // Update tracking heights/widths lastHeight = currentHeight; if (currentWidth === lastWidth) { return; // Skip graph rebuilding if it was only a keyboard height shift } lastWidth = currentWidth; clearTimeout(resizeTimeout); resizeTimeout = setTimeout(function() { renderChart(); syncDeckLayout(); }, 150); });
